### BW-CHEST-001: Standard Push-Up

```yaml
exercise:
  id: "BW-CHEST-001"
  name: "Standard Push-Up"
  arabic_name: "تمرين الضغط الأساسي"

  category:
    primary: "Chest"
    secondary: "Middle Chest"
    movement_type: "Push (Horizontal)"
    equipment_category: "Bodyweight"

  equipment:
    required: []
    optional: ["Exercise mat"]
    setup: "Clear floor space 2m x 1m. Mat optional for wrist comfort."

  difficulty:
    level: "Beginner"
    prerequisites: ["Wall Push-Up", "Incline Push-Up (if needed)"]
    progression_to: ["Diamond Push-Up", "Decline Push-Up", "Archer Push-Up"]
    regression_from: ["Knee Push-Up", "Incline Push-Up"]

  muscles:
    primary: ["Pectoralis major (sternal head)", "Pectoralis major (clavicular head)"]
    secondary: ["Anterior deltoid", "Triceps brachii"]
    stabilizers: ["Serratus anterior", "Core muscles", "Quadriceps"]

  video_production:
    recommended_angles: ["Side view (primary)", "Front 45°", "Above view"]
    key_frames: ["High plank start", "Mid-descent", "Bottom position", "Lockout"]
    slow_motion_points: ["Bottom position reversal", "Chest touching depth"]
    duration_per_rep: "2-3 seconds"
    demo_reps: "5-8 reps"

  starting_position:
    body_placement: |
      Start in high plank position. Body forms straight line from head to heels.
      Arms straight, hands on floor directly under or slightly wider than shoulders.

    foot_position: |
      Feet together or up to hip-width apart.
      Toes tucked, weight on balls of feet.
      Legs straight and engaged.

    foot_width: "Together to hip-width (narrower = harder on core)"

    knee_position: |
      Fully extended, lifted off floor.
      Kneecaps facing down, quads engaged.

    hip_position: |
      In line with shoulders and ankles.
      Slight posterior pelvic tilt (tailbone tucked).
      Glutes engaged. NO sagging or piking.

    spine_position: |
      Neutral spine - straight line.
      No lower back arch or upper back rounding.

    shoulder_position: |
      Directly over wrists (or slightly behind for more chest focus).
      Shoulder blades neutral to slightly protracted.
      Shoulders away from ears.

    arm_position: |
      Arms straight but not hyperextended.
      Hands placed shoulder-width or slightly wider.
      Elbows will bend during movement, not flared at 90°.

    hand_position: |
      Palms flat on floor, fingers spread.
      Hands turned slightly outward (comfortable angle).
      Weight distributed across whole palm.
      Hands directly under shoulders or slightly wider.

    head_position: |
      Neutral, in line with spine.
      Looking at floor about 30cm ahead of hands.
      Don't let head drop or crane up.

    core_engagement: |
      Strong brace throughout - 60-80% engagement.
      Body should feel like one rigid unit.
      Don't let abs relax during movement.

  execution:
    phases:
      - phase_name: "Descent (Eccentric)"
        movement_description: |
          Lower body toward floor by bending elbows.
          Elbows track at approximately 45° angle from torso (not flared 90°).
          Keep body rigid - everything moves together.
          Lower until chest nearly touches floor (or touches for full ROM).
          Maintain straight body line throughout descent.
        joint_actions: ["Elbow flexion", "Shoulder horizontal adduction", "Scapular retraction"]
        tempo: "2 seconds down (controlled)"
        breathing: "Inhale during descent"
        cues: ["Lower with control", "Elbows at 45°", "Chest toward floor", "Stay tight"]
        end_position: "Chest 2-5cm from floor, elbows bent approximately 90°, body still straight"

      - phase_name: "Bottom Position"
        movement_description: |
          Brief pause at bottom (optional).
          Chest at lowest point, body remains rigid.
          Do not rest on floor or relax.
        joint_actions: ["Brief isometric hold"]
        tempo: "0-1 second pause"
        breathing: "Brief hold or beginning of exhale"
        cues: ["Stay tight", "Don't relax", "Maintain position"]
        end_position: "Same as descent end"

      - phase_name: "Ascent (Concentric)"
        movement_description: |
          Push through palms to extend arms.
          Drive body away from floor as one unit.
          Maintain straight body line.
          Fully extend arms at top without hyperextending elbows.
          Shoulders protract slightly at top (push-up plus optional).
        joint_actions: ["Elbow extension", "Shoulder horizontal abduction", "Scapular protraction"]
        tempo: "1-1.5 seconds up"
        breathing: "Exhale during push (exertion)"
        cues: ["Push the floor away", "Extend arms fully", "Stay rigid", "Squeeze chest"]
        end_position: "Back to high plank starting position"

  form_cues:
    key_points:
      - "Body stays in ONE STRAIGHT LINE throughout"
      - "Elbows at 45° angle, not flared to 90°"
      - "Full range of motion - chest to floor, arms to full extension"
      - "Core braced entire time"
      - "Don't let hips sag or pike"
    mental_imagery:
      - "Imagine a broomstick along your spine - don't break contact"
      - "Push the floor away from you"
      - "Think of your body as a plank of wood"
    feeling: "Feel stretch in chest at bottom, contraction at top. Triceps and shoulders working. Core engaged."

  common_mistakes:
    - mistake: "Hips sagging"
      why_bad: "Lower back stress, easier on chest, poor form"
      correction: "Squeeze glutes, engage core harder, think 'plank position'"
      visual_indicator: "Lower back curves down, belly drops toward floor"

    - mistake: "Hips piking up"
      why_bad: "Reduces chest activation, turns into shoulder exercise"
      correction: "Lower hips to align with shoulders"
      visual_indicator: "Butt sticks up in the air"

    - mistake: "Elbows flared at 90°"
      why_bad: "Shoulder impingement risk, less chest activation"
      correction: "Keep elbows at 45° angle from torso"
      visual_indicator: "Arms form T-shape with body"

    - mistake: "Half reps (not going low enough)"
      why_bad: "Missing stretch, less muscle activation"
      correction: "Lower until chest nearly touches floor"
      visual_indicator: "Elbows barely bend, chest stays high"

    - mistake: "Head dropping"
      why_bad: "Neck strain, breaks body alignment"
      correction: "Keep head neutral, look at floor"
      visual_indicator: "Chin touches chest or head hangs"

  contraindications:
    - condition: "Shoulder injury"
      recommendation: "Modify"
      modification: "Wall push-up, incline push-up, reduce ROM"

    - condition: "Wrist pain"
      recommendation: "Modify"
      modification: "Push-up handles, fist push-ups, knuckle push-ups"

  programming:
    rep_ranges:
      strength: "5-8 reps weighted or advanced variation"
      hypertrophy: "8-15 reps"
      endurance: "15-30+ reps"
    sets: "3-4 sets"
    rest: "60-90 seconds"
    frequency: "2-4 times per week"
```

```yaml
  muscles:

    secondary: ["Anterior deltoid", "Triceps brachii (less than standard)"]
    stabilizers: ["Core muscles", "Serratus anterior"]

  video_production:
    recommended_angles: ["Above view (shows hand width)", "Side view", "Front 45°"]
    key_frames: ["Starting wide hand position", "Bottom stretch position", "Top position"]
    slow_motion_points: ["Bottom stretch (maximum chest stretch)"]
    duration_per_rep: "2-3 seconds"
    demo_reps: "5-8 reps"

  starting_position:
    body_placement: "Same as standard push-up but hands placed significantly wider."

    foot_position: "Feet together or hip-width apart, toes tucked."

    foot_width: "Hip-width or narrower"

    knee_position: "Fully extended, engaged."

    hip_position: "In line with shoulders and ankles. Glutes engaged."

    spine_position: "Neutral straight line."

    shoulder_position: "Will be higher than hands due to wide placement."

    arm_position: |
      Hands placed 1.5x to 2x shoulder width apart.
      Approximately 20-30cm wider than standard push-up on each side.
      When lowered, elbows will bend outward more than standard.

    hand_position: |
      Palms flat, fingers pointing forward or slightly outward.
      Much wider than shoulders - find comfortable max stretch width.

    head_position: "Neutral, looking at floor."

    core_engagement: "Strong brace throughout."

  execution:
    phases:
      - phase_name: "Descent (Eccentric)"
        movement_description: |
          Lower body by bending elbows outward (more than 45°).
          Feel significant stretch across chest at bottom.
          Chest travels toward floor between hands.
          Wider grip limits depth - go as low as comfortable.
        joint_actions: ["Elbow flexion", "Shoulder horizontal adduction", "Chest stretch"]
        tempo: "2-3 seconds down"
        breathing: "Inhale"
        cues: ["Feel the chest stretch", "Control descent", "Elbows bend outward"]
        end_position: "Maximum comfortable depth with chest stretch"

      - phase_name: "Ascent (Concentric)"
        movement_description: |
          Push through palms, focusing on squeezing chest together.
          Drive from outer chest fibers.
          Full arm extension at top.
        joint_actions: ["Elbow extension", "Shoulder horizontal abduction"]
        tempo: "1-2 seconds up"
        breathing: "Exhale"
        cues: ["Squeeze chest together", "Push through palms", "Full extension"]
        end_position: "Starting position"

  form_cues:
    key_points:
      - "Feel stretch across chest at bottom"
      - "Don't go so wide that shoulders hurt"
      - "Maintain body alignment"
      - "Focus on chest contraction"
    mental_imagery:
      - "Imagine hugging a large beach ball as you push up"
    feeling: "Strong chest stretch at bottom, outer chest burn during movement."

  common_mistakes:
    - mistake: "Going too wide (shoulder pain)"
      why_bad: "Shoulder impingement, injury risk"
      correction: "Find width that stretches chest without shoulder pain"
      visual_indicator: "Grimacing, shoulder shrugging, pain face"

    - mistake: "Hips sagging"
      why_bad: "Same as standard push-up"
      correction: "Engage core and glutes"
      visual_indicator: "Lower back arches down"

  contraindications:
    - condition: "Shoulder impingement or injury"
      recommendation: "Avoid"
      modification: "Standard or narrow push-up instead"

  programming:
    rep_ranges:
      hypertrophy: "8-15 reps"
      endurance: "15-25 reps"
    sets: "3-4 sets"
    rest: "60-90 seconds"
    frequency: "Include as push-up variation 1-2x per week"
```

### BW-CHEST-003: Diamond Push-Up (Close Grip)

```yaml
exercise:
  id: "BW-CHEST-003"
  name: "Diamond Push-Up"
  arabic_name: "تمرين الضغط الماسي"

  category:
    primary: "Chest"
    secondary: "Inner Chest / Triceps emphasis"
    movement_type: "Push (Horizontal - Close grip)"
    equipment_category: "Bodyweight"

  equipment:
    required: []
    optional: ["Exercise mat"]
    setup: "Clear floor space. Hands will form diamond shape under chest."

  difficulty:
    level: "Intermediate"
    prerequisites: ["Standard Push-Up (15+ reps)"]
    progression_to: ["Pseudo Planche Push-Up", "One-Arm Push-Up"]
    regression_from: ["Close Grip Push-Up (not full diamond)"]

  muscles:
    primary: ["Pectoralis major (inner/sternal emphasis)", "Triceps brachii"]
    secondary: ["Anterior deltoid"]
    stabilizers: ["Core muscles", "Serratus anterior"]

  video_production:
    recommended_angles: ["Above view (shows diamond hand position)", "Side view", "Front view"]
    key_frames: ["Diamond hand position close-up", "Bottom position", "Top lockout"]
    slow_motion_points: ["Hand position setup", "Tricep lockout"]
    duration_per_rep: "2-3 seconds"
    demo_reps: "5-8 reps"

  starting_position:
    body_placement: "High plank with hands together under chest."

    foot_position: "Feet together or slightly apart for balance."

    hand_position: |
      Hands together under chest, forming diamond/triangle shape.
      Thumbs and index fingers touch, creating diamond.
      Diamond positioned under sternum (mid-chest).
      Alternatively: hands just touching or very close (easier variation).

    arm_position: |
      Arms straight, elbows will stay close to body during movement.
      Narrower base requires more tricep and inner chest.

    core_engagement: "Extra strong brace needed - narrow base is less stable."

  execution:
    phases:
      - phase_name: "Descent (Eccentric)"
        movement_description: |
          Lower chest toward hands, keeping elbows close to ribs.
          Elbows track backward, not outward.
          Lower until chest touches hands or as low as possible.
          Significant tricep stretch at bottom.
        joint_actions: ["Elbow flexion (primary)", "Shoulder extension"]
        tempo: "2-3 seconds down"
        breathing: "Inhale"
        cues: ["Elbows stay tight to body", "Lower to the diamond", "Control descent"]
        end_position: "Chest at or near hands, elbows bent ~90° close to ribs"

      - phase_name: "Ascent (Concentric)"
        movement_description: |
          Push through hands, squeezing triceps and inner chest.
          Extend arms fully at top.
          Focus on tricep contraction through lockout.
        joint_actions: ["Elbow extension (primary)", "Shoulder flexion"]
        tempo: "1-2 seconds up"
        breathing: "Exhale"
        cues: ["Push through palms", "Squeeze triceps at top", "Lock out arms"]
        end_position: "Full arm extension, starting position"

  form_cues:
    key_points:
      - "Elbows stay close to body - don't flare"
      - "Diamond positioned under chest, not under face"
      - "Full range of motion"
      - "Control is key - this is harder than standard"
    mental_imagery:
      - "Imagine squeezing a ball between your elbows and ribs"
      - "Think of pushing yourself away from your hands"
    feeling: "Strong tricep burn, inner chest activation, more difficult than standard."

  common_mistakes:
    - mistake: "Diamond position under face instead of chest"
      why_bad: "Turns into tricep dip motion, less chest work"
      correction: "Position hands under mid-chest/sternum"
      visual_indicator: "Head moves forward past hands during descent"

    - mistake: "Elbows flaring out"
      why_bad: "Reduces tricep emphasis, defeats purpose"
      correction: "Keep elbows tucked against ribs"
      visual_indicator: "Elbows point outward during movement"

    - mistake: "Not going deep enough"
      why_bad: "Limits muscle activation"
      correction: "Touch chest to hands if possible"
      visual_indicator: "Minimal elbow bend"

  contraindications:
    - condition: "Wrist pain"
      recommendation: "Avoid or modify"
      modification: "Use push-up handles, parallel bars for close grip"

    - condition: "Elbow issues"
      recommendation: "Proceed with caution"
      modification: "Reduce ROM, don't lock out aggressively"

  programming:
    rep_ranges:
      strength: "5-8 reps"
      hypertrophy: "8-12 reps"
    sets: "3-4 sets"
    rest: "60-90 seconds"
    frequency: "Include 1-2x per week for tricep/inner chest focus"
```

```yaml
    key_points:
      - "Maintain straight body line at angle"
      - "Don't let hips sag or pike"
      - "Full range of motion"
      - "Feel it in upper chest"
    mental_imagery:
      - "Imagine pushing yourself up a hill"
      - "Think of lifting your body toward your feet"

  common_mistakes:
    - mistake: "Hips piking up"
      why_bad: "Changes angle, reduces chest work"
      correction: "Maintain straight diagonal line"
      visual_indicator: "Butt sticks up, body forms angle at hips"

    - mistake: "Not enough elevation"
      why_bad: "Same as regular push-up"
      correction: "Use 30-60cm elevation minimum"
      visual_indicator: "Body nearly horizontal"

  contraindications:
    - condition: "Shoulder issues"
      recommendation: "Proceed with caution"
      modification: "Lower elevation, standard push-up instead"

    - condition: "High blood pressure"
      recommendation: "Proceed with caution"
```

### DB-CHEST-001: Dumbbell Flat Bench Press

```yaml
exercise:
  id: "DB-CHEST-001"
  name: "Dumbbell Flat Bench Press"
  arabic_name: "تمرين ضغط الصدر بالدمبل على البنش المستوي"

  category:
    primary: "Chest"
    secondary: "Middle Chest"
    movement_type: "Push (Horizontal)"
    equipment_category: "Dumbbell"

  equipment:
    required: ["Pair of dumbbells", "Flat bench"]
    optional: ["Spotter for heavy weights"]
    setup: |
      Position flat bench in open area.
      Have dumbbells accessible on floor beside bench.
      Ensure bench is stable and locked.

  difficulty:
    level: "Beginner-Intermediate"
    prerequisites: ["Push-Up proficiency recommended"]
    progression_to: ["Incline DB Press", "Heavy DB Press", "Single-Arm DB Press"]
    regression_from: ["Push-Ups", "Machine Chest Press"]

  muscles:
    primary: ["Pectoralis major (sternal and clavicular heads)"]
    secondary: ["Anterior deltoid", "Triceps brachii"]
    stabilizers: ["Rotator cuff muscles", "Core muscles", "Serratus anterior"]

  video_production:
    recommended_angles: ["Side view (shows ROM and path)", "Above view (shows arc motion)", "Front 45°"]
    key_frames: ["Dumbbell pickup", "Starting position", "Bottom stretch", "Top contraction"]
    slow_motion_points: ["Bottom position stretch", "Pressing motion arc"]
    duration_per_rep: "3-4 seconds"
    demo_reps: "5-8 reps"

  starting_position:
    body_placement: |
      Lie on flat bench with eyes approximately under bar rack position.
      5 points of contact: head, upper back, glutes, left foot, right foot.
      Slight arch in lower back (natural lumbar curve) - not excessive.
      Shoulder blades squeezed together and down (retracted and depressed).

    foot_position: |
      Feet flat on floor, firmly planted.
      Positioned under or slightly behind knees.
      Approximately shoulder-width apart.
      Drive through feet for stability (leg drive).

    knee_position: |
      Bent at approximately 90°.
      Knees in line with hips, not caved in.

    hip_position: |
```
